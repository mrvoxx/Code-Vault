git status          //Shows changes, untracked files, and branch info

git clone https://github.com/user/repo.git           //Clone a repository from remote (git repository link) to local (to your device) location

ls         //to list all the file in that path

ls -a         //to list all the path with hidden files

git add filename         //Add a specific file (staged , ready to commit)

git add .          //Add all files (staged , ready to commit)

git commit -m "message"          //Saves your changes with a message

git remote add origin https://github.com/user/repo.git          //Add remote repository (origin == your given repository link)

git push -u origin main          //Pushes your code to the main branch for the first time.

git push          //After first time
